# Triples per graph:
SELECT ?graph (COUNT (*) AS ?triples) 
WHERE {
	GRAPH ?graph {
		?s ?p ?o
	}
} 
ORDER BY ?graph

# Count of unique enhancers (classes):
BASE <http://rdf.biogateway.eu/graph/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX obo: <http://purl.obolibrary.org/obo/>
SELECT count(distinct ?enhID)
WHERE {
        GRAPH <crm> {
                ?enhID rdfs:subClassOf obo:SO_0000727
        }
}

#Count of unique enhancers per chromosome:
BASE <http://rdf.biogateway.eu/graph/>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX obo: <http://purl.obolibrary.org/obo/>
PREFIX resource: <http://semanticscience.org/resource/>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
SELECT ?label_chr (count(distinct ?enhID) as ?count)
WHERE {
        GRAPH <crm> {
                ?enhID rdfs:subClassOf obo:SO_0000727 ;
                        obo:RO_0001025 ?chr .
                ?chr skos:prefLabel ?label_chr
        }
}
ORDER BY ?label_chr
